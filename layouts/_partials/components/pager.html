{{- /* Athenaeum override — BOOK-LINEAR prev/next pager. Hextra's default pager uses .PrevInSection/.NextInSection, which stops at chapter boundaries. The old Nextra site paged linearly across the WHOLE book (last section of a chapter → next chapter's first section). This rebuilds the book's reading order (book cover → each chapter cover → its sections → next chapter …, with book-level knowledge-checks interleaved by weight) and walks that flat list. Labels come from utils/title, same as the theme. */ -}} {{- $page := . -}} {{- $subject := $page.FirstSection -}} {{- /* the current book = the ancestor (or self) whose parent is the subject */ -}} {{- $book := "" -}} {{- range $p := ($page.Ancestors | append $page) -}} {{- with $p.Parent -}} {{- if eq .RelPermalink $subject.RelPermalink -}}{{- $book = $p -}}{{- end -}} {{- end -}} {{- end -}} {{- if $book -}} {{- /* flatten the book into reading order via composite weight keys: chapter -> "00003", its section -> "00003.00002", book-level KC -> "00007". Book cover sorts first at "00000". */ -}} {{- $flat := slice (dict "key" "00000" "page" $book) -}} {{- range $child := sort (union $book.Sections $book.RegularPages) "Weight" -}} {{- if $child.IsSection -}} {{- $ck := printf "%05d" $child.Weight -}} {{- $flat = $flat | append (dict "key" $ck "page" $child) -}} {{- range $sec := sort (union $child.RegularPages $child.Sections) "Weight" -}} {{- $flat = $flat | append (dict "key" (printf "%s.%05d" $ck $sec.Weight) "page" $sec) -}} {{- end -}} {{- else -}} {{- $flat = $flat | append (dict "key" (printf "%05d" $child.Weight) "page" $child) -}} {{- end -}} {{- end -}} {{- $flat = sort $flat "key" -}} {{- $idx := -1 -}} {{- range $i, $e := $flat -}} {{- if eq $e.page.RelPermalink $page.RelPermalink -}}{{- $idx = $i -}}{{- end -}} {{- end -}} {{- $prev := "" -}} {{- $next := "" -}} {{- if ge $idx 0 -}} {{- if gt $idx 0 -}}{{- $prev = (index $flat (sub $idx 1)).page -}}{{- end -}} {{- if lt $idx (sub (len $flat) 1) -}}{{- $next = (index $flat (add $idx 1)).page -}}{{- end -}} {{- end -}} {{- if or $prev $next -}}
