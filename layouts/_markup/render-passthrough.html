{{- /* Athenaeum Populi — math passthrough render hook (overrides Hextra's). Supports the Phase 1 math benchmark by rendering KaTeX EITHER at build time OR client-side, chosen by site.Params.math.render ("build" | "client"). build -> transform.ToMath (KaTeX runs in Hugo; page ships static HTML). client -> emit the raw math wrapped in \(...\) / \[...\] delimiters and let the browser's KaTeX auto-render (loaded in custom/head-end.html) do the work. Either way we set Page.Store "hasMath" so head.html loads the KaTeX CSS, and "mathClient" so head-end.html knows to inject the client-side renderer. */ -}} {{- $engine := site.Params.math.engine | default "katex" -}} {{- $mode := site.Params.math.render | default "build" -}} {{- if and (eq $engine "katex") (eq $mode "build") -}} {{- $opts := dict "output" "htmlAndMathml" "displayMode" (eq .Type "block") -}} {{- with try (transform.ToMath .Inner $opts) -}} {{- with .Err -}} {{- errorf "transform.ToMath failed to render math (%s): %s" $.Position . -}} {{- else -}} {{- .Value -}} {{- $.Page.Store.Set "hasMath" true -}} {{- end -}} {{- end -}} {{- else -}} {{- /* client-side KaTeX (or mathjax): keep the source, add delimiters back */ -}} {{- $.Page.Store.Set "hasMath" true -}} {{- if eq $engine "katex" -}} {{- $.Page.Store.Set "mathClient" true -}} {{- end -}} {{- if eq .Type "block" -}} \[{{- .Inner -}}\] {{- else -}} \({{- .Inner -}}\) {{- end -}} {{- end -}}
